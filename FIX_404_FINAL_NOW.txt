ВЫПОЛНИТЕ НА СЕРВЕРЕ ПРЯМО СЕЙЧАС для исправления 404:

# 1. Включите mod_rewrite (если еще не включен)
sudo a2enmod rewrite

# 2. Обновите файлы из репозитория
cd /ssd/www/videoeditor
git pull origin main

# 3. Обновите конфигурацию Apache
sudo cp config/apache.conf /etc/apache2/sites-available/videoeditor.conf

# 4. Установите правильные права
sudo chown -R www-data:www-data /ssd/www/videoeditor
sudo chmod -R 755 /ssd/www/videoeditor
sudo chmod 644 /ssd/www/videoeditor/public/.htaccess
sudo chmod 644 /ssd/www/videoeditor/public/index.php

# 5. Проверьте конфигурацию Apache
sudo apache2ctl configtest

# 6. Перезапустите Apache
sudo systemctl restart apache2

# 7. Проверьте что mod_rewrite включен
apache2ctl -M | grep rewrite

# 8. Создайте тестовый файл для проверки
sudo tee /ssd/www/videoeditor/public/test.php << 'EOF'
<?php
echo "TEST OK\n";
echo "REQUEST_URI: " . ($_SERVER['REQUEST_URI'] ?? 'NOT SET') . "\n";
EOF
sudo chmod 644 /ssd/www/videoeditor/public/test.php
sudo chown www-data:www-data /ssd/www/videoeditor/public/test.php

# 9. Проверьте логи
sudo tail -50 /var/log/apache2/videoeditor_ssl_error.log

# 10. Проверьте в браузере:
# https://videoeditor.1tlt.ru/test.php - должен показать REQUEST_URI
# https://videoeditor.1tlt.ru/index.php - должен открыть приложение
# https://videoeditor.1tlt.ru/login - должен открыть страницу входа
