ВЫПОЛНИТЕ НА СЕРВЕРЕ для исправления 500 ошибки:

# 1. Уберите FallbackResource из конфигурации Apache
sudo nano /etc/apache2/sites-available/videoeditor.conf

# Найдите секцию <Directory /ssd/www/videoeditor/public>
# Уберите строку: FallbackResource /index.php
# Оставьте только:
# <Directory /ssd/www/videoeditor/public>
#     Options -Indexes +FollowSymLinks
#     AllowOverride All
#     Require all granted
#     
#     # PHP-FPM через mod_proxy_fcgi
#     <FilesMatch \.php$>
#         SetHandler "proxy:fcgi://127.0.0.1:9000"
#     </FilesMatch>
# </Directory>

# Сохраните (Ctrl+O, Enter, Ctrl+X)

# 2. Убедитесь что mod_rewrite включен
sudo a2enmod rewrite

# 3. Проверьте конфигурацию
sudo apache2ctl configtest

# 4. Перезапустите Apache
sudo systemctl restart apache2

# 5. Проверьте что .htaccess настроен правильно
cat /ssd/www/videoeditor/public/.htaccess | grep -A 5 "RewriteEngine"

# 6. Проверьте логи
sudo tail -50 /var/log/apache2/videoeditor_ssl_error.log

# 7. Проверьте в браузере:
# https://videoeditor.1tlt.ru/login - должно работать!
